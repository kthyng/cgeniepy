    import xarray as xr
    import cgeniepy
    import numpy as np

    model = cgeniepy.sample_model()

    sst = model.get_var('ocn_sur_temp').isel(time=-1)

    ## grid category is a pre-defined 2D data
    ## 0: coastal
    ## 1: land
    ## 2: open ocean
    gc = model.grid_category()

    ## coastal region
    xr.where(gc == 0, sst.data.values, np.nan).plot()
